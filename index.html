/** * Google Apps Script - 試験管理アプリ用 (v3) * * セットアップ手順: * 1. Googleスプレッドシートを新規作成 * 2. 「拡張機能」→「Apps Script」を開く * 3. このコードを貼り付けて保存 * 4. 「デプロイ」→「新しいデプロイ」→ 種類:「ウェブアプリ」 * - 実行するユーザー: 「自分」 * - アクセスできるユーザー: 「全員」 * 5. 生成されたURLを試験アプリの管理画面に貼り付ける * * ※重要: コードを更新したら必ず「新しいデプロイ」で新バージョンを作成してください */ // ===== スプレッドシート取得ヘルパー ===== function getSpreadsheet() { // まずgetActiveを試す（コンテナバインドスクリプトの場合） var ss = SpreadsheetApp.getActiveSpreadsheet(); if (ss) return ss; // ダメならスクリプトプロパティからIDを取得 var props = PropertiesService.getScriptProperties(); var ssId = props.getProperty('spreadsheetId'); if (ssId) { try { return SpreadsheetApp.openById(ssId); } catch(e) { // IDが無効な場合は次へ } } // 最終手段: このスクリプトの親ファイル（スプレッドシート）を探す var scriptId = ScriptApp.getScriptId(); var file = DriveApp.getFileById(scriptId); var parents = file.getParents(); while (parents.hasNext()) { var parent = parents.next(); var files = parent.getFiles(); while (files.hasNext()) { var f = files.next(); if (f.getMimeType() === 'application/vnd.google-apps.spreadsheet') { var id = f.getId(); props.setProperty('spreadsheetId', id); return SpreadsheetApp.openById(id); } } } // それでもダメなら新規作成 ss = SpreadsheetApp.create('試験結果'); props.setProperty('spreadsheetId', ss.getId()); return ss; } // POST: 回答を受け取ってスプレッドシートに保存 function doPost(e) { try { var data; // フォーム送信とJSON送信の両方に対応 if (e.parameter && e.parameter.payload) { data = JSON.parse(e.parameter.payload); } else { data = JSON.parse(e.postData.contents); } var action = data.action || 'submit'; // 問題データの保存 if (action === 'saveQuestions') { var props = PropertiesService.getScriptProperties(); props.setProperty('examSettings', JSON.stringify(data.settings || {})); props.setProperty('examQuestions', JSON.stringify(data.questions || [])); return ContentService.createTextOutput(JSON.stringify({ status: 'ok', message: 'Questions saved' })) .setMimeType(ContentService.MimeType.JSON); } // スプレッドシートIDの登録（管理者が手動設定する場合） if (action === 'setSpreadsheetId') { var props = PropertiesService.getScriptProperties(); props.setProperty('spreadsheetId', data.spreadsheetId); return ContentService.createTextOutput(JSON.stringify({ status: 'ok', message: 'Spreadsheet ID saved' })) .setMimeType(ContentService.MimeType.JSON); } // 回答の送信 var ss = getSpreadsheet(); var sheet = ss.getSheetByName('Sheet1') || ss.getSheets()[0]; if (sheet.getLastRow() === 0) { var headers = ['タイムスタンプ', '名前', '受験番号', '得点', '満点', '正答率(%)']; var ansCount = data.answers ? data.answers.split('|||').length : 0; for (var i = 1; i <= ansCount; i++) { headers.push('問' + i); } sheet.appendRow(headers); sheet.getRange(1, 1, 1, headers.length).setFontWeight('bold'); } var row = [ data.timestamp || new Date().toISOString(), data.name || '', data.id || '', data.score || 0, data.totalPoints || 0, data.percentage || 0 ]; if (data.answers) { var answers = data.answers.split('|||'); answers.forEach(function(a) { row.push(a); }); } sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({ status: 'ok', message: 'Answer saved' })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } } // GET: スプレッドシートのデータを取得（集計用） function doGet(e) { try { var action = (e.parameter && e.parameter.action) ? e.parameter.action : ''; // 問題データの取得 if (action === 'getQuestions') { var props = PropertiesService.getScriptProperties(); var settings = props.getProperty('examSettings'); var questions = props.getProperty('examQuestions'); return ContentService.createTextOutput(JSON.stringify({ status: 'ok', settings: settings ? JSON.parse(settings) : {}, questions: questions ? JSON.parse(questions) : [] })).setMimeType(ContentService.MimeType.JSON); } // 回答データの取得 if (action === 'getAll') { var ss = getSpreadsheet(); var sheet = ss.getSheetByName('Sheet1') || ss.getSheets()[0]; var data = sheet.getDataRange().getValues(); if (data.length <= 1) { return ContentService.createTextOutput(JSON.stringify({ submissions: [] })) .setMimeType(ContentService.MimeType.JSON); } var submissions = []; for (var i = 1; i < data.length; i++) { var row = data[i]; var answers = []; for (var j = 6; j < row.length; j++) { answers.push({ questionIndex: j - 5, answer: String(row[j]), correct: false }); } submissions.push({ timestamp: row[0], name: row[1], id: row[2], score: row[3], totalPoints: row[4], percentage: row[5], answers: answers }); } return ContentService.createTextOutput(JSON.stringify({ submissions: submissions })) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ status: 'ok', message: 'Exam API is running (v3)' })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })) .setMimeType(ContentService.MimeType.JSON); } }
